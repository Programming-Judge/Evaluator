# Evaluator

# Setup:
    bash setup.bash

# Run:
    cd ../../src
    go run .

# Add a problem:
    localhost:7070/uploadProblem?fileName=1.tar.gz

# Add a submission:
    localhost:7070/uploadSubmission?fileName=1.tar.gz

# Remove a problem: 
    localhost:7070/rmProblem?problemID=1

# Remove a submission: 
    localhost:7070/rmSubmission?ID=1
